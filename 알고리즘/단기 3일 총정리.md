    for(int i = 2; i <= n; i++) {
        if (!check[i])
        {
            // i가 소수임, 소수 출력이라면 i를 출력

        }

        }
    }

## 삼각 달팽이

    vector<vector<int>> MAP(n + 1, vector<int>(n + 1));
    vector<int> answer;
    int Max_Num = (n * (n + 1)) / 2;
    int Top = 1, Bottom = n, Left = 1, Right = 0;
    int Num = 1, State = 0;
    
    while (Num <= Max_Num)
    {
        if (State == 0)
        {
            for (int i = Top; i <= Bottom; i++)
            {
                MAP[i][Left] = Num++;
            }
            
            Top++;
            Left++;
            State = 1;
        }
        else if (State == 1)
        {
            for (int i = Left; i <= Bottom - Right; i++)
            {
                MAP[Bottom][i] = Num++;
            }
            
            Bottom--;
            State = 2;
        }
        else if (State == 2)
        {
            for (int i = Bottom; i >= Top; i--)
            {
                MAP[i][i - Right] = Num++;
            }
            
            Right++;
            Top++;
            State = 0;
        }
    }
